** Protobuf preparation

#+BEGIN_SRC proto :tangle pb/crypto.proto
syntax = "proto2";

package crypto.pb;

enum KeyType {
	RSA = 0;
	Ed25519 = 1;
	Secp256k1 = 2;
	ECDSA = 3;
}

message PublicKey {
	required KeyType Type = 1;
	required bytes Data = 2;
}

message PrivateKey {
	required KeyType Type = 1;
	required bytes Data = 2;
}
#+END_SRC

#+BEGIN_SRC sh
protoc -I ./pb --python_out=./src ./pb/crypto.proto
#+END_SRC
